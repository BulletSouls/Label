CREATE TABLE Grad(
GradID INT NOT NULL PRIMARY KEY,
ImeGrada VARCHAR (255) NOT NULL,
PozivniBroj VARCHAR (255) NOT NULL,
PostanskiBroj VARCHAR (255) NOT NULL,
BrojStanovnika INT NOT NULL,
LokalitetID INT NOT NULL,
DrzavaID INT NOT NULL,
CONSTRAINT FK_LokalitetID FOREIGN KEY (LokalitetID)
REFERENCES Lokalitet(LokalitetID),
CONSTRAINT FK_DrzavaID FOREIGN KEY (DrzavaID)
REFERENCES Drzava(DrzavaID)
);
